```javascript
function changeTShirtColor(color: string) {
    const mesh = scene.getObjectByName('tshirt') as THREE.Mesh;
    if(mesh) {
      (mesh.material as THREE.MeshBasicMaterial).color.set(color);
    }
}
```

![image.png](https://p9-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/d40403a1256e4e68bd367373ef9df991~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=1412&h=986&s=202447&e=png&b=1f1f1f)

```javascript
interface TshirtRef {
  changeTShirtColor: (color: string) => void
}
const Tshirt = forwardRef<TshirtRef>((props, ref) => {
```
```javascript
useImperativeHandle(ref, () => {
    return {
      changeTShirtColor
    }
})
```
forwardRef 来实现 ref 的转发，useImperativeHandle 来做 ref 的自定义。

然后在父组件里用 useRef 来创建 ref 接收：

```javascript
interface ColorListProps {
  onColorChange: (color: string) => void
}
function ColorList(props: ColorListProps) {
  const colors = [
    'red', 'orange', 'pink',
    'lightgreen', 'lightblue', 'yellow', 
    'purple', 'grey', 'white',
    'green', 'cyan', 'blue'
  ]

  return <div className='color-list'>
    {
      colors.map(color => {
        return <div
          key={color}
          className='color-item'
          style={{background: color}}
          onClick={() => props.onColorChange(color)}
        ></div>
      })
    }
  </div>
}
```
