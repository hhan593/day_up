调用 HTTP 时，返回速度可能较慢，所以我们把 UserInfo 和 RepoInfo 加入 Redis 缓存。

use axum::http::StatusCode;
use axum::response::{IntoResponse, Response};
use thiserror::Error;

// 为了使用方便，可以用 HelloResult<T> 代替 Result<T, HelloError>
pub type HelloResult<T> = Result<T, HelloError>;

#[derive(Error, Debug)]
pub enum HelloError {
    #[error("not implemented")]
    NotImplemented,
    #[error("not found")]
    NotFound,
    // #[from] 表示自动 impl From<std::io::Error> for HelloError
    // 如果错误类型是 srd::io::Error，通过 ? 返回错误时可以自动转换成 HelloError
    #[error("io error: {0}")]
    IO(#[from] std::io::Error),
    #[error("db error: {0}")]
    Db(#[from] sea_orm::DbErr),
    #[error("forbidden")]
    Forbidden,
    #[error("reqwest error: {0}")]
    Reqwest(#[from] reqwest::Error),
    #[error("redis error: {0}")]
    Redis(#[from] redis::RedisError),
}

impl IntoResponse for HelloError {
    fn into_response(self) -> Response {
        let code = match self {
            Self::NotImplemented => StatusCode::NOT_IMPLEMENTED,
            Self::NotFound => StatusCode::NOT_FOUND,
            Self::IO(_) => StatusCode::INTERNAL_SERVER_ERROR,
            Self::Db(_) => StatusCode::INTERNAL_SERVER_ERROR,
            Self::Forbidden => StatusCode::FORBIDDEN,
            Self::Reqwest(_) => StatusCode::INTERNAL_SERVER_ERROR,
            Self::Redis(_) => StatusCode::INTERNAL_SERVER_ERROR,
        };
        let body = self.to_string();
        (code, body).into_response()
    }
}

# 读写 redis 的方法

接下来会介绍4种读写 Redis 的方法：

  


1.  使用 set 前，把 UserInfo 转为 Json String。使用 get 后，把 Json String 解析为 UserInfo

    1.  好处：简单、容易想到，如果 UserInfo 定义在第三方库中也可以使用
    1.  坏处：多一次内存分配

```Rust
use crate::error::HelloResult;
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize, Clone, Debug, Default)]
pub struct UserInfo {
    pub login: String,
    pub followers: i32,
    pub following: i32,
    pub public_repos: i32,
}

pub async fn get_user_info(username: String) -> HelloResult<UserInfo> {
    use redis::AsyncCommands;
    let key = format!("user:{}", username);
    if let Some(s) = crate::redis::CLIENT
        .get_async_connection()
        .await?
        .get::<&str, Option<String>>(&key)
        .await?
    {
        let user_info = serde_json::from_str(&s).unwrap();
        return Ok(user_info);
    }
    let client = reqwest::Client::builder()
        .user_agent("hello-axum")
        .build()?;
    let url = format!("https://api.github.com/users/{}", username);
    let user_info = client.get(url).send().await?.json().await?;
    crate::redis::CLIENT
        .get_async_connection()
        .await?
        .set(&key, serde_json::to_string(&user_info).unwrap())
        .await?;
    Ok(user_info)
}
```

  


2.  为 UserInfo 实现 `redis::FromRedisValue` 和 `redis::ToRedisArgs`

    1.  好处：减少内存分配
    1.  坏处：每个 Struct 都需要自己实现 `redis::FromRedisValue` 和 `redis::ToRedisArgs`，比较麻烦。不能为依赖库中的类型实现不是自己定义的 trait；

> 根据孤儿原则（orphan rule），如果 Struct 和 Trait 都不是自己定义的，不能为 Struct 实现 Trait。

```Rust
use crate::error::HelloResult;
use redis::{RedisResult, RedisWrite, Value};
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize, Clone, Debug, Default)]
pub struct UserInfo {
    pub login: String,
    pub followers: i32,
    pub following: i32,
    pub public_repos: i32,
}

impl redis::ToRedisArgs for UserInfo {
    fn write_redis_args<W>(&self, out: &mut W)
    where
        W: ?Sized + RedisWrite,
    {
        out.write_arg(serde_json::to_string(&self).unwrap().as_bytes())
    }
}

impl redis::FromRedisValue for UserInfo {
    fn from_redis_value(v: &Value) -> RedisResult<Self> {
        match *v {
            Value::Data(ref bytes) => serde_json::from_slice(bytes)
                .map_err(|_| redis::RedisError::from((redis::ErrorKind::ResponseError, ""))),
            _ => Err(redis::RedisError::from((
                redis::ErrorKind::TypeError,
                "Response type not demo_data compatible",
            ))),
        }
    }
}

pub async fn get_user_info(username: String) -> HelloResult<UserInfo> {
    use redis::AsyncCommands;
    let key = format!("user:{}", username);
    if let Some(user_info) = crate::redis::CLIENT
        .get_async_connection()
        .await?
        .get(&key) // .get::<&str, Option<UserInfo>>(&key) 这里可以自动推断类型，不需要类型标注了
        .await?
    {
        return Ok(user_info); // 直接返回，不需要解析 json string了
    }
    let client = reqwest::Client::builder()
        .user_agent("hello-axum")
        .build()?;
    let url = format!("https://api.github.com/users/{}", username);
    let user_info = client.get(url).send().await?.json().await?;
    crate::redis::CLIENT
        .get_async_connection()
        .await?
        .set(&key, &user_info) // 直接传入 user_info，不需要先转 json string
        .await?;
    Ok(user_info)
}
```

  


3.  编写宏，自动实现 `redis::FromRedisValue` 和 `redis::ToRedisArgs`，类似`#[derive(Default)]`

    1.  好处：减少内存分配，使用方便
    1.  坏处：不能为依赖库的类型实现 trait，学习难度比较大

比较复杂，这里不展开介绍，学习资料：[proc-macro-workshop](https://github.com/dtolnay/proc-macro-workshop)

建议 B 站或 YouTube 搜索 "proc macro workshop" 跟着视频学习

  


4.  定义 Wrapper 类型，为 Wrapper 类型实现 `redis::FromRedisValue` 和 `redis::ToRedisArgs`

    1.  好处：减少内存分配，使用比较方便，如果 Struct 定义在第三方库中也可以用。

use lazy_static::lazy_static;
use redis::{FromRedisValue, RedisError, RedisResult, RedisWrite, ToRedisArgs, Value};
use serde::de::DeserializeOwned;
use serde::Serialize;

lazy_static! {
    pub static ref CLIENT: redis::Client =
        redis::Client::open("redis://:123456@localhost/0").unwrap();
}

// 因为 impl Trait for Struct 的时候，Trait/Struct 至少要有一个定义在当前 mod
// 所以定义 RedisJsonData，对其他 Struct 进行包装
pub struct RedisJsonData<T>(pub T);

impl<T: DeserializeOwned> FromRedisValue for RedisJsonData<T> {
    fn from_redis_value(v: &Value) -> RedisResult<Self> {
        match *v {
            Value::Data(ref bytes) => serde_json::from_slice(bytes)
                .map(RedisJsonData)
                .map_err(|_| RedisError::from((redis::ErrorKind::ResponseError, ""))),
            _ => Err(RedisError::from((
                redis::ErrorKind::TypeError,
                "Response type not demo_data compatible",
            ))),
        }
    }
}

impl<T: Serialize> ToRedisArgs for RedisJsonData<T> {
    fn write_redis_args<W>(&self, out: &mut W)
    where
        W: ?Sized + RedisWrite,
    {
        out.write_arg(serde_json::to_string(&self.0).unwrap().as_bytes())
    }
}

pub struct UserInfo {
    pub login: String,
    pub followers: i32,
    pub following: i32,
    pub public_repos: i32,
}

pub async fn get_user_info(username: String) -> HelloResult<UserInfo> {
    let key = format!("user:{}", username);

    // 读缓存，如果存在直接返回
    // 模式匹配语法，和之前的 handler 参数中的 Json(req): Json<T> 一样
    if let Some(RedisJsonData(info)) = crate::redis::CLIENT
        .get_async_connection()
        .await?
        .get(&key) // .get::<&str, Option<RedisJsonData<UserInfo>>>(&key)
        .await?
    {
        return Ok(info);
    }

    // 请求 api
    let client = reqwest::Client::builder()
        .user_agent("hello-axum")
        .build()?;
    let url = format!("https://api.github.com/users/{}", username);
    let info = client.get(url).send().await?.json().await?;

    // 写缓存
    crate::redis::CLIENT
        .get_async_connection()
        .await?
        .set(&key, RedisJsonData(&info))
        .await?;
    Ok(info)
}
