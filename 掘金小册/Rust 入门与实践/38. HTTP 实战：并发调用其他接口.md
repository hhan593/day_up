```Rust
use axum::http::StatusCode;
use axum::response::{IntoResponse, Response};
use thiserror::Error;

// 为了使用方便，可以用 HelloResult<T> 代替 Result<T, HelloError>
pub type HelloResult<T> = Result<T, HelloError>;

#[derive(Error, Debug)]
pub enum HelloError {
    #[error("not implemented")]
    NotImplemented,
    #[error("not found")]
    NotFound,
    // #[from] 表示自动 impl From<std::io::Error> for HelloError
    // 如果错误类型是 srd::io::Error，通过 ? 返回错误时可以自动转换成 HelloError
    #[error("io error: {0}")]
    IO(#[from] std::io::Error),
    #[error("db error: {0}")]
    Db(#[from] sea_orm::DbErr),
    #[error("forbidden")]
    Forbidden,
    #[error("reqwest error: {0}")]
    Reqwest(#[from] reqwest::Error),
}

impl IntoResponse for HelloError {
    fn into_response(self) -> Response {
        let code = match self {
            Self::NotImplemented => StatusCode::NOT_IMPLEMENTED,
            Self::NotFound => StatusCode::NOT_FOUND,
            Self::IO(_) => StatusCode::INTERNAL_SERVER_ERROR,
            Self::Db(_) => StatusCode::INTERNAL_SERVER_ERROR,
            Self::Forbidden => StatusCode::FORBIDDEN,
            Self::Reqwest(_) => StatusCode::INTERNAL_SERVER_ERROR,
        };
        let body = self.to_string();
        (code, body).into_response()
    }
}
```

# 创建 github handler

> 这种并发方案不适用于计算密集型任务。
>
> -   相同类型的任务用 stream
> -   不同类型的任务用 join!

  


创建 src/handler/github.rs

```Rust
use crate::api::github::{get_repo_info, get_user_info, RepoInfo, UserInfo};
use crate::error::HelloResult;
use axum::Json;
use futures::stream::StreamExt;
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize, Clone, Debug, Default)]
pub struct GithubInfoRequest {
    users: Vec<String>,
    repos: Vec<String>,
}

#[derive(Serialize, Deserialize, Clone, Debug, Default)]
pub struct GithubInfoResponse {
    user_infos: Vec<Option<UserInfo>>,
    repo_infos: Vec<Option<RepoInfo>>,
}

pub async fn github_info(
    Json(req): Json<GithubInfoRequest>,
) -> HelloResult<Json<GithubInfoResponse>> {
    // 同时执行多个相同类型的任务
    // 1. 把 req.users 转换为 stream
    // 2. 利用 .map，把 String 转为 Future
    // 3. buffered 收集 Future 结果
    // 4. collect 变成 Vector
    // 可以直接 .await 得到 Vec<HelloResult<UserInfo>>，这里先不 .await ，下面用 join! 和另一个 future 一起执行
    let user_infos_future = futures::stream::iter(req.users)
        .map(get_user_info) // 等价于 .map(|s| get_user_info(s))
        .buffered(10) // 最多同时10并发
        .collect();

    let repo_infos_future = futures::stream::iter(req.repos)
        .map(get_repo_info) // 等价于 .map(|s| get_repo_info(s))
        .buffered(10) // 最多同时10并发
        .collect();

    // 同时执行两个不同的 future
    let (user_infos, repo_infos): (Vec<HelloResult<UserInfo>>, Vec<HelloResult<RepoInfo>>) =
        tokio::join!(user_infos_future, repo_infos_future);

    // 把 Error 变成 None，也就是 null
    let user_infos = user_infos.into_iter().map(|info| info.ok()).collect();
    let repo_infos = repo_infos.into_iter().map(|info| info.ok()).collect();

    Ok(Json(GithubInfoResponse {
        user_infos,
        repo_infos,
    }))
}
```

# 添加 handler 到 router

```Rust
use axum::routing::{get, post};
use axum::{Router, Server};
use hello_axum::handler::check::{must_login, no_login, optional_login};
use hello_axum::handler::github::github_info;
use hello_axum::handler::user::{login, register};
use hello_axum::{hello, not_found_handler, not_implemented_handler, read_file_handler};
use std::net::SocketAddr;

#[tokio::main]
async fn main() {
    tracing_subscriber::fmt::init();

    let app = Router::new()
        .route("/hello", get(hello))
        .nest(
            "/error",
            Router::new()
                .route("/not_implemented", get(not_implemented_handler))
                .route("/read_file", get(read_file_handler)),
        )
        .nest(
            "/user",
            Router::new()
                .route("/register", post(register))
                .route("/login", post(login)),
        )
        .nest(
            "/check",
            Router::new()
                .route("/must_login", get(must_login))
                .route("/optional_login", get(optional_login))
                .route("/no_login", get(no_login)),
        )
        .route("/github_info", post(github_info))
        .fallback(not_found_handler);

    // run it
    let addr = SocketAddr::from(([0, 0, 0, 0], 3000));
    tracing::info!("listening on {addr}");
    Server::bind(&addr)
        .serve(app.into_make_service())
        .await
        .unwrap();
}
```

# 其他并发方案

下面会介绍一些其他的并发方案（tokio::spawn + 锁/队列），在此之前需要了解一些注意事项。

在 async fn 中一定要使用 async 相关的锁、队列。（比如用 `tokio::sync::Mutex`，不能用 `std::sync::Mutex`）

  


不要在 async fn、tokio::spawn 中做以下的事：

1.  使用非 async await 语法的锁、队列（std 的也不行）
1.  执行高强度计算任务
1.  执行同步 IO 操作
1.  其他耗时较长，并且不带 .await 的操作

  


如果需要执行高强度计算任务、同步 IO 操作、或其他耗时较长非 async 的函数，建议在 tokio 的 [.spawn_blocking](https://docs.rs/tokio/latest/tokio/task/fn.spawn_blocking.html) 中执行，通过 JoinHandle 或队列把结果传回原来的 task。

  


原因（如果不能理解，记住上面的就够了）：

async 的关键在于 runtime 执行 task 时，遇到 .await 如果是 Pending 状态，会让出执行权给其他 task。如果执行任务中没有 .await（比如纯计算、死循环、标准库锁），runtime 不会切换到其他任务执行，也就会导致线程卡死。

  


1.  创建 `Arc<Mutex<Result>>`，[tokio::spawn](https://docs.rs/tokio/latest/tokio/task/fn.spawn.html#examples) 多个线程(task)执行代码，拿到结果后，给 result 加锁写入。

>**常规套路**
>
>-   线程间共享需要修改的变量：`Arc<Mutex<T>>`/`Arc<RwLock<T>>`，传递给新线程之前 clone，使用前 `.lock().await`/`.read().await`/`.write.await`，使用完尽快 drop
>-   线程间共享不需要修改的变量：Arc<T>，传递给新线程之前 clone

```Rust
use std::sync::Arc;
use tokio::sync::Mutex;

let results = Arc::new(Mutex::new(vec![]));
let mut handles = vec![];
for i in 0..3 {
    handles.push(tokio::spawn({
        let results = results.clone();
        async move {
            results.lock().await.push(i);
            // let mut results = results.lock().await; // 加锁
            // results.push(i);
            // drop(results); // 解锁
        }
    }));
}
for handle in handles {
    handle.await.unwrap();
}
println!("{:?}", results.lock().await);
```


2.  创建 channel，[tokio::spawn](https://docs.rs/tokio/latest/tokio/task/fn.spawn.html#examples) 多个线程(task)执行代码，拿到结果后，通过 channel 发送结果

```Rust
let mut results = vec![];
let mut receivers = vec![];
for i in 0..3 {
    let (sender, receiver) = tokio::sync::oneshot::channel();
    receivers.push(receiver);
    tokio::spawn(async move {
        sender.send(i).unwrap();
    });
}
for receiver in receivers {
    results.push(receiver.await.unwrap());
}
println!("{:?}", results);
```
  


3.  利用 tokio::spawn 的返回值 JoinHandle<T>

```Rust
let mut results = vec![];
let mut handles = vec![];
for i in 0..3 {
    handles.push(tokio::spawn(async move {
        return i;
    }));
}
for handle in handles {
    results.push(handle.await.unwrap());
}
println!("{:?}", results);
```
