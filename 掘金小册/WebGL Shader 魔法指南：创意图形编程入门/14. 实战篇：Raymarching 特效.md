## 创建平面

新建`RaymarchingScene.js`文件。

```sh
touch src/Experience/World/RaymarchingScene.js
```

在里面导出`RaymarchingScene`类。

```js
import * as kokomi from "kokomi.js";

export default class RaymarchingScene extends kokomi.Component {
  constructor(base) {
    super(base);
  }
}
```

`Raymarching`场景的一切都是在一张铺满屏幕的`2D`平面上渲染的。

`kokomi.js`有个内置的`ScreenQuad`类，能够创建出这样的平面。

```js
import * as kokomi from "kokomi.js";

export default class RaymarchingScene extends kokomi.Component {
  constructor(base) {
    super(base);

    this.quad = new kokomi.ScreenQuad(this.base);
  }
  addExisting() {
    this.quad.addExisting();
  }
}
```

在`World.js`中，我们导入`RaymarchingScene`类并初始化它。

```js
import RaymarchingScene from "./RaymarchingScene";

export default class World extends kokomi.Component {
  constructor(base) {
    super(base);

    this.base.am.on("ready", () => {
      this.rs = new RaymarchingScene(this.base);
      this.rs.addExisting();
      document.querySelector(".loader-screen")?.classList.add("hollow");
    });
  }
}
```

画面上会显示一张铺满屏幕的`UV`坐标图，这个是`ScreenQuad`类默认的着色器所渲染的结果，我们要用自己的着色器来代替它。

在`Shaders`目录下新建`RaymarchingScene`目录，并在该目录下新建 1 个`Shader`文件：片元着色器`frag.glsl`。

```sh
mkdir src/Experience/Shaders/RaymarchingScene

```

把以下的着色器代码直接复制进`frag.glsl`。

```glsl
void mainImage(out vec4 fragColor,in vec2 fragCoord){
    vec2 uv=fragCoord/iResolution.xy;
    fragColor=vec4(uv,0.,1.);
}
```

在`RaymarchingScene.js`中，我们引入这个着色器，作为`ScreenQuad`的参数，并且把`shadertoyMode`设为`true`。

```js
import raymarchingSceneFragmentShader from "../Shaders/RaymarchingScene/frag.glsl";

export default class RaymarchingScene extends kokomi.Component {
  constructor(base) {
    super(base);

    this.quad = new kokomi.ScreenQuad(this.base, {
      fragmentShader: raymarchingSceneFragmentShader,
      shadertoyMode: true,
    });
  }
  addExisting() {
    this.quad.addExisting();
  }
}
```

画面显示的结果还是跟刚刚一样，但我们已经可以开始自定义着色器了。

## 基础场景

之前我们写`Raymarching`场景的时候要复制粘贴一大堆现成的函数代码，但这里就不用了，我们有`lygia`这一神器在。

```glsl
#define RESOLUTION iResolution.xy// 分辨率
#define RAYMARCH_SAMPLES 128// 采样次数（即步进次数）
#define RAYMARCH_MULTISAMPLE 1// 多重采样次数（即抗锯齿次数）
#define RAYMARCH_BACKGROUND vec3(0.)// 背景色
#define RAYMARCH_CAMERA_FOV 2.// 相机视场角

#include "/node_modules/lygia/space/ratio.glsl"

#define RAYMARCH_MATERIAL_FNC raymarchCustomMaterial
vec3 raymarchCustomMaterial(vec3 ray,vec3 pos,vec3 nor,vec3 map);

#include "/node_modules/lygia/lighting/raymarch.glsl"
#include "/node_modules/lygia/sdf.glsl"

vec3 raymarchCustomMaterial(vec3 ray,vec3 pos,vec3 nor,vec3 map){
    vec3 col=vec3(0.);

    col+=map*.2;

    vec3 lightPos=vec3(10.);
    vec3 lightDir=normalize(lightPos-pos);
    float diff=max(dot(lightDir,nor),0.);
    col+=map*diff;

    vec3 reflectDir=reflect(-lightDir,nor);
    vec3 viewDir=normalize(-ray);
    vec3 halfVec=normalize(lightDir+viewDir);
    float spec=pow(max(dot(nor,halfVec),0.),32.);
    col+=map*spec;

    return col;
}

vec4 raymarchMap(vec3 p){
    vec4 res=vec4(1.);
    res=opUnion(res,vec4(vec3(.875,.286,.333),sphereSDF(p-vec3(0.,0.,-2.),1.)));
    return res;
}

void mainImage(out vec4 fragColor,in vec2 fragCoord){
    vec2 uv=fragCoord/iResolution.xy;
    uv=ratio(uv,iResolution.xy);
    vec3 col=vec3(0.);
    vec3 camera=vec3(0.,0.,30.);
    col=raymarch(camera,uv).rgb;
    fragColor=vec4(col,1.);
}
```

开头我们用宏定义了一些配置用的参数，含义用注释标出来了。

然后，我们引入了一些`lygia`内置的工具函数：`ratio`能修正画面的比例，`raymarch`封装了整个光线步进的过程，`sdf`封装了`SDF`相关的所有函数，在`raymarch`函数的上方我们声明了`raymarchCustomMaterial`函数，用来自定义光照和材质，在下方我们将该函数具体实现一下（步骤跟之前[第十章](https://juejin.cn/book/7267462574734573604/section/7267463336034795556)的光照部分几乎一样）。

接着，我们需要定义一个存放所有`SDF`的`map`函数，这里`lygia`把这个函数命名为了`raymarchMap`，并且返回了一个四维的数据。向场景中添加物体依旧用的是“并”操作`opUnion`函数，但它的第二个参数也变成了四维数据：前三个维度代表了物体本身的颜色值，最后一个维度代表了物体的`SDF`函数输出的距离，`SDF`函数名由之前的`sdSphere`变成了`sphereSDF`，参数不变。

> 想了解所有的`SDF`函数定义的话，可以看`lygia`的`sdf`部分的[源码](https://github.com/patriciogonzalezvivo/lygia/tree/main/sdf)。

最后，在主函数中，我们用`ratio`修正了画面比例，定义了相机位置`camera`，将它和`uv`一同传给`raymarch`函数，再将结果输出到屏幕上。

经过分析，我们可以将这个物体大致分成这几个部分：头部、耳朵、杯体、裙边、花纹（觉得难以实现的部分可以选择性地忽略）；主要颜色有白色、红色、黄色、黑色。

然后，我们就可以开始准备一些代码相关的东西了：建造这个模型需要几种`SDF`函数？那些函数的具体参数是什么？主要颜色所对应的具体变量值是什么？

### SDF 函数

对于`SDF`函数，我们可以参照下`iq`的这个所有`SDF`函数的`demo`：

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/2336e7cece724541a4ef7aea65009fa7~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=274240&e=jpg&b=928c94)

> `demo`地址：https://www.shadertoy.com/view/Xds3zN

通过观察，我们能初步确定需要这些`SDF`函数：球体`sdSphere`、椭圆体`sdEllipsoid`、带圆角的圆锥体`sdRoundCone`。此外，还有一个万能的方块函数`sdBox`（主要用于布尔运算）。

由于我们用的是`lygia`框架，我们还需要在它的[源码](https://github.com/patriciogonzalezvivo/lygia/tree/main/sdf)里面找到对应的函数（这里省略了函数内部的具体实现，我们主要关注参数）：

```glsl
float sphereSDF(vec3 p,float s){...}

float ellipsoidSDF(in vec3 p,in vec3 r){...}

float coneSDF(in vec3 p,in float r1,float r2,float h){...}

float boxSDF(vec3 p,vec3 b){...}
```

`sphereSDF`和`ellipsoidSDF`基本跟`iq`的一致，最后一个`coneSDF`有几种变体，我们选择了其中带圆角的这个变体。

来看看这些函数的参数分别代表什么吧：第一个`p`不用多说，肯定是位置`position`。`sphereSDF`的第 2 个参数之前已经说过了，代表球体的半径。那其他没见过的`SDF`函数的参数含义呢？全是缩写，连个注释都没有，一时半会还真猜不出来。

这时候，我们就要回到上面的那个 [demo](https://www.shadertoy.com/view/Xds3zN)，找到调用那些函数的位置，亲自动手来改变那些参数的值，再观察图形有哪些变化，下图是具体的操作步骤：

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/a4c47aac1964411babdd76cef9da69ce~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=447&s=431772&e=jpg&b=e5e3e3)

> 注：左边的画面是支持用鼠标拖动来进行轨道观察的。

通过这个方法，我们就能归纳出这些函数参数的含义：

```glsl
/*
球体的SDF函数
s: 球体半径
*/
float sphereSDF(vec3 p,float s){...}

/*
椭圆体的SDF函数
r: 椭圆体大小，x代表长度，y代表高度，z代表厚度
*/
float ellipsoidSDF(in vec3 p,in vec3 r){...}

/*
圆锥的SDF函数
r1: 底部半径
r2: 顶部半径
h：高度
*/
float coneSDF(in vec3 p,in float r1,float r2,float h){...}

/*
方块的SDF函数
b：方块大小，x代表长度，y代表高度，z代表厚度
*/
float boxSDF(vec3 p,vec3 b){...}
```

> 鉴于我们之前已经通过`lygia`的`sdf.glsl`引入过了所有的`SDF`函数，你无需在`Shader`文件里将它们再次声明一遍。不过你也可以将它们的本体注释掉，放在`Shader`的顶部，这样就能随时查阅它们的含义。

当然，在`Shader`中，我们是无法直接使用这个六进制值的，要把它转换成`vec3`形式的变量。为了做到这一点，我封装好了一个`JS`函数：

```js
// https://stackoverflow.com/a/5624139
const hexToRgb = (hex) => {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result
    ? {
        r: parseInt(result[1], 16),
        g: parseInt(result[2], 16),
        b: parseInt(result[3], 16),
      }
    : null;
};

const toFixed3 = (num) => num.toFixed(3);

const getGLSLColorFromHex = (color) => {
  const rgb = hexToRgb(color);
  return `vec3(${toFixed3(rgb.r / 255)},${toFixed3(rgb.g / 255)},${toFixed3(
    rgb.b / 255
  )})`;
};
```

按`F12`打开控制台，将上面的函数直接粘贴上去运行，再把你拾取到的颜色值作为参数来调用`getGLSLColorFromHex`这个函数，你就能得到`vec3`形式的颜色变量。

把所有的主要颜色都拾取、转换后，我把它们作为宏定义放在了`Shader`的顶部，方便之后的使用。（有的颜色值还可以根据个人喜好进行适当地微调。）

## 建模

新建一个`jumpyDumpty`函数，它就是我们的模型函数。

```glsl
vec4 jumpyDumpty(vec3 p,vec4 res){
    return res;
}

vec4 raymarchMap(vec3 p){

    // res=opUnion(res,vec4(vec3(.875,.286,.333),sphereSDF(p-vec3(0.,0.,-2.),1.)));
```

### 头部

先从头部开始建吧，新增一个球体。

```glsl
vec4 jumpyDumpty(vec3 p,vec4 res){
    // head
    vec3 p1=p;
    float head=sphereSDF(p1,.69);
    res=opUnion(res,vec4(COLOR_1,head));

    return res;
}
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/f694aa3578a24ca19a1f6477cf6cc285~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=18017&e=jpg&b=010101)

在球体的上方创建一个方块，并且要和球体保持一部分的重合。

```glsl
vec4 jumpyDumpty(vec3 p,vec4 res){
    // head
    vec3 p1=p;
    float head=sphereSDF(p1,.69);
    head=opUnion(head,boxSDF(p1-vec3(0.,1.3,0.),vec3(1.)));
    res=opUnion(res,vec4(COLOR_1,head));

    return res;
}
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/2ea7859d50b6456d98403f75b3eeaa56~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=43329&e=jpg&b=020202)

然后，我们把“并”操作改成“交”操作，取方块和球体相交的部分。

```glsl
vec4 jumpyDumpty(vec3 p,vec4 res){
    // head
    vec3 p1=p;
    float head=sphereSDF(p1,.69);
    head=opIntersection(head,boxSDF(p1-vec3(0.,1.3,0.),vec3(1.)));
    res=opUnion(res,vec4(COLOR_1,head));

    return res;
}
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/ecb5b536eee444a1a2bc3e28702f23ea~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=10326&e=jpg&b=000000)

这样我们相当于用方块把球体的上面部分给“切割”了下来，得到了头部。

### 耳朵

耳朵的形状很明显是一个椭圆体，对应的`SDF`函数是`ellipsoidSDF`。

```glsl
vec4 jumpyDumpty(vec3 p,vec4 res){
    ...

    // ear
    vec3 p2=p;
    float ear=ellipsoidSDF(p2-vec3(.45,.7,0.),vec3(.1,.25,.1));
    res=opUnion(res,vec4(COLOR_1,ear));

    return res;
}
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/40d8c00bc3604cf9bd5ebebb109754da~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=11960&e=jpg&b=000000)

目前它的朝向不对，我们需要用旋转函数`rotate`来把它旋转至正确的方向。

`lygia`内置的`rotate`函数接收 3 个参数：第一个是旋转点`p`，第二个是旋转角度大小，第三个是旋转轴，我们将椭球体绕着`z`轴顺时针旋转个 60 度吧。

```glsl
vec4 jumpyDumpty(vec3 p,vec4 res){
    ...

    // ear
    vec3 p2=p;
    float ear=ellipsoidSDF(rotate(p2-vec3(.45,.7,0.),-PI/3.,vec3(0.,0.,1.)),vec3(.1,.25,.1));
    res=opUnion(res,vec4(COLOR_1,ear));

    return res;
}
```

耳朵看上去有点“嵌”在头部里面，我们可以调整`opUnion`函数的第 3 个参数，将耳朵和头部给“粘”在一起。

```glsl
vec4 jumpyDumpty(vec3 p,vec4 res){
    ...
```

```glsl
vec3 raymarchCustomMaterial(vec3 ray,vec3 pos,vec3 nor,vec3 map){
    vec3 col=vec3(0.);

    col+=map*.2;

    vec3 lightPos=vec3(10.);
    vec3 lightDir=normalize(lightPos-pos);
    float diff=max(dot(lightDir,nor),0.);
    col+=map*diff;

    vec3 reflectDir=reflect(-lightDir,nor);
    vec3 viewDir=normalize(-ray);
    vec3 halfVec=normalize(lightDir+viewDir);
    float spec=pow(max(dot(nor,halfVec),0.),32.);
    col+=map*spec;

    if(map==COLOR_1){
        // head

        // eye
        vec2 uv1=pos.xy;
        uv1.x=abs(uv1.x);
        uv1/=vec2(.75,.5);
        uv1-=vec2(-.15,.4);
        float c1=circleSDF(uv1);
        float eye=1.-smoothstep(.15,.151,c1);
        col=mix(col,COLOR_3,eye);

        // mouth
        vec2 uv2=pos.xy;
        uv2.x=abs(uv2.x);
        uv2.y-=.4;
        uv2.y*=-1.;
        float c2=lineSDF(uv2,vec2(0.),vec2(.05));
        float mouth=1.-smoothstep(.0125,.01251,c2);
        col=mix(col,COLOR_3,mouth);
    }else if(map==COLOR_4){
        // belt
        vec2 uv3=pos.xy;
        uv3/=vec2(.4);
        uv3-=vec2(-.5,-.18);
        uv3=fract(uv3);
        float c3=circleSDF(uv3);
        float polka=1.-smoothstep(.15,.151,c3);
        col=mix(col,COLOR_5,polka);
    }else if(map==COLOR_2){
        // body

        vec2 uv4=pos.xy;
        uv4/=vec2(.4);
        uv4-=vec2(-.5,.05);
        uv4.y/=.6;
        uv4.y*=-1.;
        float c4=triSDF(uv4);
        float tri=1.-smoothstep(1.8,1.81,c4);
        col=mix(col,COLOR_6,tri);

        vec2 uv5=pos.xy;
        uv5/=vec2(.4);
        uv5-=vec2(-.5,-1.05);
        float c5=uv5.y;
        float stripe=smoothstep(.09,.1,c5)-smoothstep(.19,.2,c5);
        col=mix(col,COLOR_5,stripe);

        vec2 uv6=pos.xy;
        uv6/=vec2(.4);
        uv6-=vec2(-.7,-1.9);
        uv6.y/=.7;
        uv6.x/=1.4;
        float c6=heartSDF(uv6);
        float heart=1.-smoothstep(1.8,1.81,c6);

    }

    return col;
}
```

## 场景效果

`Raymarching`有个很神奇的功能：它能够无限地复制一个物体。

`lygia`已经帮我们封装好了这个函数，但新版的命名可能和旧版的不太一样，我们用宏定义来统一一下。

```glsl
#include "/node_modules/lygia/sdf.glsl"
#ifndef opRepeat
#define opRepeat opRepite
#endif
```

函数的内部实现是这样的（无需将它复制进代码）：

```glsl
vec3 opRepeat(in vec3 p,in vec3 c){
    return mod(p+.5*c,c)-.5*c;
}
```

其中`mod`函数是求余函数，它能获取除法计算的余数，比如`mod(7,3)`会返回 1，它的函数图像如下（以`mod(x,3)`为例）：

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/fc42a07f346447b18a5151c65b3bbf62~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=721&s=108399&e=jpg&b=212121)

它也是一个具有重复性质的函数，之前学的获取数字的小数部分的`fract`函数就等价于`mod(x,1)`。

继续来看上面的`opRepeat`函数，它的主体是`mod(p,c)`，c 代表间隔，此外还多了一个居中的操作：先加上一半的间隔，再减去一半的间隔。

我们在主函数里对模型函数的点`p1`使用重复函数吧。

```glsl
vec4 raymarchMap(vec3 p){
    ...

    vec3 p1=p;
    p1=opRepeat(p1,vec3(3.));
    res=jumpyDumpty(p1,res);

    return res;
}
```

```glsl

    p.x-=1.5;
    p.z-=iTime*2.;
    p.y-=abs(sin(iTime*5.))*.2;
    p=opRepeat(p,vec3(3.));
    return p;
}
```

为了防止左下角的蹦蹦离我们太近，给`x`轴分量减去了一定的偏移量。然后给`z`轴分量减去`iTime`使得蹦蹦能够往前移动，给`y`轴分量减去`sin`和`abs`共同作用下的`iTime`能使蹦蹦产生周期性的跳跃。
