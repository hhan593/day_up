## 随机函数

在`JS`中，我们可以用它的内置函数`Math.random`来模拟随机，但是，直到本小册成书为止，`GLSL`却仍没有内置的随机函数可供使用。所幸的是，在`GitHub`上，已经有人封装好了随机函数——[glsl-random](https://github.com/mattdesl/glsl-random)，先来看看它是什么样的。

```glsl
highp float random(vec2 co)
{
    highp float a=12.9898;
    highp float b=78.233;
    highp float c=43758.5453;
    highp float dt=dot(co.xy,vec2(a,b));
    highp float sn=mod(dt,3.14);
    return fract(sin(sn)*c);
}
```

这个随机函数的意义是取`sin`函数偏后面的小数部分来模拟伪随机数。

我们可以给它传入一个`UV`坐标，看看会输出什么效果。

```glsl
void mainImage(out vec4 fragColor,in vec2 fragCoord){
    vec2 uv=fragCoord/iResolution.xy;

    float noise=random(uv);

    vec3 col=vec3(noise);
    fragColor=vec4(col,1.);
}
```

## 噪声

噪声也是实现“随机”的一种手段，只不过它能使实现的效果更加偏向“自然”一些。

平时我们在生活中看到的云雾、岩石、火焰等，都有噪声的存在。

到本小册成书为止，跟缺少内置的随机函数一样，`GLSL`里同样也没有内置的噪声函数。但是在`GitHub`上，也已经有人封装好了一些常用的噪声函数——[glsl-noise](https://github.com/hughsk/glsl-noise)。我们选择其中比较常用的柏林噪声（`Perlin Noise`）的`3D`版本，看看它是什么样的。

```glsl
//
// GLSL textureless classic 3D noise "cnoise",
// with an RSL-style periodic variant "pnoise".
// Author:  Stefan Gustavson (stefan.gustavson@liu.se)
// Version: 2011-10-11
//
// Many thanks to Ian McEwan of Ashima Arts for the
// ideas for permutation and gradient selection.
//
// Copyright (c) 2011 Stefan Gustavson. All rights reserved.
// Distributed under the MIT license. See LICENSE file.
// https://github.com/ashima/webgl-noise
//

vec3 mod289(vec3 x)
{
    return x-floor(x*(1./289.))*289.;
}

vec4 mod289(vec4 x)
{
    return x-floor(x*(1./289.))*289.;
}

vec4 permute(vec4 x)
{
    return mod289(((x*34.)+1.)*x);
}

vec4 taylorInvSqrt(vec4 r)
{
    return 1.79284291400159-.85373472095314*r;
}

vec3 fade(vec3 t){
    return t*t*t*(t*(t*6.-15.)+10.);
}

// Classic Perlin noise
float cnoise(vec3 P)
{
    vec3 Pi0=floor(P);// Integer part for indexing
    vec3 Pi1=Pi0+vec3(1.);// Integer part + 1
    Pi0=mod289(Pi0);
    Pi1=mod289(Pi1);
    vec3 Pf0=fract(P);// Fractional part for interpolation
    vec3 Pf1=Pf0-vec3(1.);// Fractional part - 1.0
    vec4 ix=vec4(Pi0.x,Pi1.x,Pi0.x,Pi1.x);
    vec4 iy=vec4(Pi0.yy,Pi1.yy);
    vec4 iz0=Pi0.zzzz;
    vec4 iz1=Pi1.zzzz;

    vec4 ixy=permute(permute(ix)+iy);
    vec4 ixy0=permute(ixy+iz0);
    vec4 ixy1=permute(ixy+iz1);

    vec4 gx0=ixy0*(1./7.);
    vec4 gy0=fract(floor(gx0)*(1./7.))-.5;
    gx0=fract(gx0);
    vec4 gz0=vec4(.5)-abs(gx0)-abs(gy0);
    vec4 sz0=step(gz0,vec4(0.));
    gx0-=sz0*(step(0.,gx0)-.5);
    gy0-=sz0*(step(0.,gy0)-.5);

    vec4 gx1=ixy1*(1./7.);
    vec4 gy1=fract(floor(gx1)*(1./7.))-.5;
    gx1=fract(gx1);
    vec4 gz1=vec4(.5)-abs(gx1)-abs(gy1);
    vec4 sz1=step(gz1,vec4(0.));
    gx1-=sz1*(step(0.,gx1)-.5);
    gy1-=sz1*(step(0.,gy1)-.5);

    vec3 g000=vec3(gx0.x,gy0.x,gz0.x);
    vec3 g100=vec3(gx0.y,gy0.y,gz0.y);
    vec3 g010=vec3(gx0.z,gy0.z,gz0.z);
    vec3 g110=vec3(gx0.w,gy0.w,gz0.w);
    vec3 g001=vec3(gx1.x,gy1.x,gz1.x);
    vec3 g101=vec3(gx1.y,gy1.y,gz1.y);
    vec3 g011=vec3(gx1.z,gy1.z,gz1.z);
    vec3 g111=vec3(gx1.w,gy1.w,gz1.w);

    vec4 norm0=taylorInvSqrt(vec4(dot(g000,g000),dot(g010,g010),dot(g100,g100),dot(g110,g110)));
    g000*=norm0.x;
    g010*=norm0.y;
    g100*=norm0.z;
    g110*=norm0.w;
    vec4 norm1=taylorInvSqrt(vec4(dot(g001,g001),dot(g011,g011),dot(g101,g101),dot(g111,g111)));
    g001*=norm1.x;
    g011*=norm1.y;
    g101*=norm1.z;
    g111*=norm1.w;

    float n000=dot(g000,Pf0);
    float n100=dot(g100,vec3(Pf1.x,Pf0.yz));
    float n010=dot(g010,vec3(Pf0.x,Pf1.y,Pf0.z));
    float n110=dot(g110,vec3(Pf1.xy,Pf0.z));
    float n001=dot(g001,vec3(Pf0.xy,Pf1.z));
    float n101=dot(g101,vec3(Pf1.x,Pf0.y,Pf1.z));
    float n011=dot(g011,vec3(Pf0.x,Pf1.yz));
    float n111=dot(g111,Pf1);

    vec3 fade_xyz=fade(Pf0);
    vec4 n_z=mix(vec4(n000,n100,n010,n110),vec4(n001,n101,n011,n111),fade_xyz.z);
    vec2 n_yz=mix(n_z.xy,n_z.zw,fade_xyz.y);
    float n_xyz=mix(n_yz.x,n_yz.y,fade_xyz.x);
    return 2.2*n_xyz;
}
```

尽管代码有点长，但我们只要会用它就行，给它传入`UV`，看看会输出什么。

## FBM

`FBM`全称是`Fractal Brownian Motion`，中译是“分形布朗运动”，它是将多个具有不同频率和振幅的噪声的结果值相叠加而产生结果的一种随机过程。它主要被用于构造自然界的云层、山脉、地貌等不规则的形体（本文开头的雨林景象用的就是这种技术）。

上面我们用的是柏林噪声，在这里我把噪声换成了更加自然的`Simplex`噪声，它的具体代码就不贴了，直接从[这里](https://github.com/hughsk/glsl-noise/blob/master/simplex/3d.glsl)拷过来吧。

然后我们来实现`fbm`函数。

```glsl
float fbm(vec3 p){
    float value=0.;
    float amplitude=1.;
    float frequency=1.;
    float lacunarity=2.;
    float persistance=.5;
    float scale=1.;
    int octaves=8;

    for(int i=0;i<octaves;i++){
        float noiseVal=snoise(p*frequency*scale);

        value+=amplitude*noiseVal;
        frequency*=lacunarity;
        amplitude*=persistance;
    }

    return value;
}
```

先定义几个变量：`value`代表结果值、`amplitude`代表振幅、`frequency`代表频率，`lacunarity`代表空隙、`persistance`代表持续度、`scale`代表缩放程度、`octaves`代表音度。

然后写个`for`循环来叠加噪声，叠加次数就是音度，在叠加的同时升高频率，降低振幅。

函数实现后，我们用`UV`作为参数来观察它的输出。

```glsl
void mainImage(out vec4 fragColor,in vec2 fragCoord){
    vec2 uv=fragCoord/iResolution.xy;
    float noise=fbm(vec3(uv,0.));
    vec3 col=vec3(noise);
    fragColor=vec4(col,1.);
}
```

那这种消融效果是怎么实现的呢？我们可以用刚学的`FBM`来试试。

在`Shader`最上方引入 2 张纹理。


创建消融函数`dissolve`。

```glsl
vec4 getFromColor(vec2 uv){
    return texture(iChannel0,uv);
}

vec4 getToColor(vec2 uv){
    return texture(iChannel1,uv);
}

vec4 dissolve(vec2 uv){
    float progress=iMouse.x/iResolution.x;
    float ratio=iResolution.x/iResolution.y;

    vec4 col=getFromColor(uv);
    vec4 col1=getToColor(uv);

    return col;
}
```

在`mainImage`函数中应用它。

```glsl
void mainImage(out vec4 fragColor,in vec2 fragCoord){
    vec2 uv=fragCoord/iResolution.xy;

    vec4 col=dissolve(uv);

    fragColor=col;
}
```

在`dissolve`函数中，复制一份`UV`出来，命名为`p`，居中它，用它来创建`fbm`噪声。

```glsl
vec2 p=uv;
p-=.5;
p.x*=ratio;
float noise=fbm(vec3(p,0.));
```

用`smoothstep`函数将`progress`与`noise`相比较，`progress`从左侧开始是 0，小于`noise`时，`pr`会返回 1，鼠标从左侧慢慢右移时，`progress`会慢慢变大，`pr`也会慢慢变成小于 1 的值。

```glsl
float pr=smoothstep(progress-.01,progress,noise);
col.rgb=mix(col.rgb,col1.rgb,1.-pr);
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/62ea084be914431ea11cae66180ea96a~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=503296&e=jpg&b=938d6e)

> 这里如果觉得效果有点卡顿，可以考虑把`fbm`函数里的`octaves`改小点，比如改成 2。

然而，当我们把鼠标拖到最左侧时，会发现已经出现了噪声，这是因为目前噪声的值并不在 0 到 1 之间变化，为了让噪声的值域接近`[0,1]`，我们需要调整噪声的值。

```glsl
noise=(noise*.5)+.5;
```

调整后，我们会发现结果依旧还是有一点误差。为了消除这个误差，我们要引入一个新的函数——`remap`，该函数的功能是将一个值的值域重新映射到另一个值域上。

```glsl
float remap(float a,float b,float c,float d,float t)
{
    return clamp((t-a)/(b-a),0.,1.)*(d-c)+c;
}
```

在这里，我们将噪声的值域从原先近似的`[-1,1]`重新映射到值域`[0,1]`。

```glsl
// noise=(noise*.5)+.5;
noise=remap(-1.,1.,0.,1.,noise);
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/149209eca3574a6d9938cbc7cf7e065e~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=3345106&e=gif&f=59&b=7f7a64)

不错，这样就实现了基本的消融效果。

此外，消融时还有一圈亮边，我们也来试着实现它吧。

先定义 2 个变量：亮边的长度和颜色。

```glsl
float edgeWidth=.1;
vec3 edgeColor=vec3(0.835,0.694,0.051);
```

绘制亮边的遮罩，长度是`edgeWidth`，混合度是噪声与位置的差，然后将遮罩输出到画布上看看。

```glsl
float edge=1.-smoothstep(0.,edgeWidth,noise-progress);
col.rgb=vec3(edge);
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/dfaef197645942b28e8e814e4c5988da~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=2164214&e=gif&f=82&b=000000)

根据`smoothstep`函数的定义，当`noise-progress`小于 0，也就是`noise`小于`progress`时，`smoothstep`会返回 0，`edge`会返回 1，此时鼠标应该会在画布的右侧，画布整体变成了白色，表示消融变化后；反之，当`noise-progress`大于 `edgeWidth`，也就是`noise`大于`progress`时，`smoothstep`会返回 1，`edge`会返回 0，此时鼠标应该会在画布的左侧，画布整体变成了黑色，表示消融变化前；当`noise-progress`在 0 到`edgeWidth`之间时，正好处在消融变化之中。

理解后，我们将输出颜色的那一行`col.rgb=vec3(edge);`注释掉，将亮边的遮罩作为混合因素，亮边颜色作为目标色，与纹理的颜色混合起来。

```glsl
// col.rgb=vec3(edge);
col.rgb=mix(col.rgb,edgeColor,edge);
```

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/60ecb418bd9e40158553a5fc435fed3a~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=540&s=313971&e=jpg&b=d0b00f)

亮边确实出现了，但是之前的消融部分哪去了呢？那是因为消融部分的调用顺序在前面，跑到亮边的下面一层去了，而亮边本身的范围要比消融大，就把它给遮盖掉了。为了解决这个问题，我们将两者的调用顺序对调一下吧。

```glsl
float edgeWidth=.1;
vec3 edgeColor=vec3(.835,.694,.051);
float edge=1.-smoothstep(0.,edgeWidth,noise-progress);
// col.rgb=vec3(edge);
col.rgb=mix(col.rgb,edgeColor,edge);

float pr=smoothstep(progress-.01,progress,noise);
col.rgb=mix(col.rgb,col1.rgb,1.-pr);
```

现在层级对了，然而还有一个可以优化的点，我们将鼠标拖到最左侧。

![](https://p3-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/f1bfc1b82a0c47d9be745aa0bf7603dd~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=960&h=447&s=344011&e=jpg&b=868266)

可以发现在左侧时本不该出现的亮边也出现了，这是因为此时的`progress`为 0，`noise-progress`的值为`noise`，而`noise`的值不一定会大于亮边长度`edgeWidth`，导致了`edge`不一定等于 0。我们的目的是要确保在左侧时混合因素的值为 0。

```glsl
col.rgb=mix(col.rgb,edgeColor,edge*step(.0001,progress));
```

我们给混合因素乘上一个值，当`progress`小于一个很小的值（如`0.0001`）时，这个值会返回 0，否则就返回 1，这样就能确保鼠标位于最左侧时没有多余的亮边出现了。

## 小结

来简要地总结下本文的要点吧：

1. 随机函数`random`是一种实现“随机”的方式。通过它，我们能得到带有噪点的图形效果。
2. 噪声`noise`也是实现“随机”的方式，并且得到的结果更加“自然”。我们可以将噪声叠加到一些规则的图形上，让它变成不规则的图形。
3. 分形布朗运动`FBM`是将多个具有不同频率和振幅的噪声的结果值相叠加而产生结果的一种随机过程，常用于模拟自然界的山脉等不规则形体。我们可以用`FBM`来实现一种“消融”的效果。

随机，可以说是各种炫酷效果必不可少的一个组成部分，一旦用好，定会为你的效果增色不少。
