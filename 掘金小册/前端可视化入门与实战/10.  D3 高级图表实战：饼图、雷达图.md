```js
const pie = d3.layout.pie()
```

可以看到，我们通过 `d3.layout.pie()` 创建一个饼图布局，并将其存储在 `pie` 变量中。使用 `.sort(null)` 方法来禁用对扇形的排序。如果传入一个比较函数作为参数，可以根据数据的某个属性对扇形进行排序。最后使用 `.value()` 方法定义了每个扇形的值。这个方法接收一个函数作为参数，该函数用于从数据对象中提取出用于绘制饼图的值。

通过调用 `d3.svg.arc()` 函数创建了另一个名为 `outerArc` 的弧形生成器，并设置了内半径和外半径。我们通过这个生成外环和 label。

然后我们定义了一个颜色比例尺 `color` ，使用了 D3.js 的 `scale.ordinal()` 方法来创建一个离散的比例尺。 `domain()` 方法定义了比例尺的输入域，即标签数组，包括了"Rose"、"Sunflower"、"Tulip"等等。 `range()` 方法定义了比例尺的输出范围，即颜色数组，包括了不同的颜色值。

最后通过 randomData 生成动态的数据给 change 方法处理。


### 创建环形

```js
function change(data) {
    /* 生成环形 */
    const slice = svg.select(".slices").selectAll("path.slice")
        .data(pie(data), key);

    slice.enter()
        .insert("path")
        .style("fill", function (d) { return color(d.data.label); })
        .attr("class", "slice");

    slice
        .transition().duration(1000)
        .attrTween("d", function (d) {
            this._current = this._current || d;
            const interpolate = d3.interpolate(this._current, d);
            this._current = interpolate(0);
            return function (t) {
                return arc(interpolate(t));
            };
        })

    slice.exit()
        .remove();
        
    // ....
}
```
首先，我们通过 `svg.select(".slices")` 选择了类名为"slices"的元素，并使用 `selectAll("path.slice")` 选择了所有类名为"slice"的路径元素。然后，将数据绑定到这些路径元素上，使用了 `data(pie(data), key)` 。其中， `pie(data)` 是将数据转换为饼图布局， `key` 是一个函数，用于指定数据的唯一标识。 

接下来，使用 `slice.enter()` 选择新加入的数据，并通过 `insert("path")` 在每个数据点上插入一个新的路径元素。使用 `style("fill", function (d) { return color(d.data.label); })` 为每个路径元素设置填充颜色，颜色根据数据的标签使用之前定义的颜色比例尺 `color` 来确定。  

然后，通过 `attr("class", "slice")` 为路径元素添加类名。

接着，使用 `slice.transition().duration(1000)` 方法来设置路径元素的过渡效果，持续时间为1秒。

在过渡效果中，使用 `attrTween("d", function (d) { ... })` 方法来设置路径的插值动画。在这个函数中，使用 `d3.interpolate()` 方法创建一个插值函数，用于计算当前路径和目标路径之间的中间状态。然后，通过 `arc(interpolate(t))` 函数将插值结果转换为路径字符串。 

最后，使用 `slice.exit().remove()` 方法将多余的路径元素移除。



### 生成标签

```js
/* 生成标注 */
const text = svg.select(".labels").selectAll("text")
    .data(pie(data), key);

text.enter()
    .append("text")
    .attr("dy", ".35em")
    .text(function (d) {
        return d.data.label;
    });

function midAngle(d) {
    return d.startAngle + (d.endAngle - d.startAngle) / 2;
}

text.transition().duration(1000)
    .attrTween("transform", function (d) {
        this._current = this._current || d;
        const interpolate = d3.interpolate(this._current, d);
        this._current = interpolate(0);
        return function (t) {
            const d2 = interpolate(t);
            const pos = outerArc.centroid(d2);
            pos[0] = radius * (midAngle(d2) < Math.PI ? 1 : -1);
            return "translate(" + pos + ")";
        };
    })
    .styleTween("text-anchor", function (d) {
        this._current = this._current || d;
        const interpolate = d3.interpolate(this._current, d);
        this._current = interpolate(0);
        return function (t) {
            const d2 = interpolate(t);
            return midAngle(d2) < Math.PI ? "start" : "end";
        };
    });

text.exit()
    .remove();

```

首先我们生成所有的文本元素 `.labels`，使用 `.data(pie(data), key)` 将数据绑定到文本元素上，`pie(data)` 将数据转换为饼图布局。 

然后，我们使用 `.enter()` 方法来获取新添加的文本元素，并使用 `.append("text")` 方法将它们添加到 `.labels` 元素中。 `.attr("dy", ".35em")` 设置文本的垂直对齐方式为中心对齐。   

接下来，我们使用 `.text(function (d) { return d.data.label; })` 将文本内容设置为数据对象的"label"属性。

`midAngle()` 函数用于计算每个数据对象的中间角度。 然后，我们使用 `.transition().duration(1000)` 方法来创建一个过渡效果，并使用 `.attrTween("transform", function (d) { ... })` 方法来实现文本元素的动画移动。 在 `.attrTween("transform", function (d) { ... })` 函数中，我们首先保存当前的数据对象到 `this._current` ，然后使用 `d3.interpolate()` 函数创建一个插值函数 `interpolate` ，用于计算当前数据对象和目标数据对象之间的中间状态。返回的函数接收一个时间参数 `t` ，并根据插值函数的结果计算当前时间点的位置。在这个函数中，我们使用 `interpolate(t)` 计算当前时间点的插值结果，并使用 `outerArc.centroid(d2)` 计算文本的位置。 `pos[0]` 根据中间角度的位置确定文本的水平方向。 

接下来，我们使用 `.styleTween("text-anchor", function (d) { ... })` 方法来实现文本的动画对齐方式。 在 `.styleTween("text-anchor", function (d) { ... })` 函数中，我们使用与上述相同的方法来保存当前的数据对象到 `this._current` ，然后使用 `d3.interpolate()` 函数创建一个插值函数 `interpolate` ，用于计算当前数据对象和目标数据对象之间的中间状态。返回的函数接收一个时间参数 `t` ，并根据插值函数的结果计算当前时间点的对齐方式。根据中间角度的位置，如果小于 π，则对齐方式为"start"，否则为"end"。 

最后，我们通过 `.exit().remove()` 方法来移除不再需要的文本元素。



### 生成引线


```js
/* 生成引线 */
const polyline = svg.select(".lines").selectAll("polyline")
    .data(pie(data), key);

polyline.enter()
    .append("polyline");

polyline.transition().duration(1000)
    .attrTween("points", function (d) {
        this._current = this._current || d;
        const interpolate = d3.interpolate(this._current, d);
        this._current = interpolate(0);
        return function (t) {
            const d2 = interpolate(t);
            const pos = outerArc.centroid(d2);
            pos[0] = radius * 0.95 * (midAngle(d2) < Math.PI ? 1 : -1);
            return [arc.centroid(d2), outerArc.centroid(d2), pos];
        };
    });

polyline.exit()
    .remove();
```

在 `.attrTween("points", function (d) { ... })` 函数中，我们首先保存当前的数据对象到 `this._current` ，然后使用 `d3.interpolate()` 函数创建一个插值函数 `interpolate` ，用于计算当前数据对象和目标数据对象之间的中间状态。返回的函数接收一个时间参数 `t` ，并根据插值函数的结果计算当前时间点的位置。

在这个函数中，我们使用 `interpolate(t)` 计算当前时间点的插值结果，并使用 `arc.centroid(d2)` 和 `outerArc.centroid(d2)` 计算折线的起点和终点位置。 `pos[0]` 根据中间角度的位置确定折线的`水平方向`。 

最后，我们使用 `.exit().remove()` 方法来移除不再需要的折线元素。

### 生成多边形

```js
// 生成多边形
let series = 0;

d.forEach(function (y, x) {
    let dataValues = [];
    y.forEach(function (j, i) {
        let value = Math.max(j.value, 0);
        let angle = i * cfg.radians / total;
        let xCoord = cfg.w / 2 * (1 - (parseFloat(value) / cfg.maxValue) * cfg.factor * Math.sin(angle));
        let yCoord = cfg.h / 2 * (1 - (parseFloat(value) / cfg.maxValue) * cfg.factor * Math.cos(angle));
        dataValues.push([xCoord, yCoord]);
    });
    dataValues.push(dataValues[0]);

    g.append("polygon")
        .data([dataValues])
        .attr("class", "radar-chart-serie" + x)
        .style("stroke-width", "2px")
        .style("stroke", cfg.color(x))
        .style("fill", cfg.color(x))
        .style("fill-opacity", cfg.opacityArea)
        .attr("points", function (d) {
            return d.map(function (point) {
                return point[0] + "," + point[1];
            }).join(" ");
        })
        .on('mouseover', function () {
            g.selectAll("polygon")
                .transition(200)
                .style("fill-opacity", 0.1);
            d3.select(this)
                .transition(200)
                .style("fill-opacity", 0.7);
        })
        .on('mouseout', function () {
            g.selectAll("polygon")
                .transition(200)
                .style("fill-opacity", cfg.opacityArea);
        });

    series++;
});
```

首先，使用变量 `series` 初始化为 0，用于给每个多边形设置唯一的类名。使用 `forEach` 方法遍历数组 `d` 中的每个元素，并传入回调函数。回调函数中的参数 `y` 表示数组元素的值， `x` 表示数组元素的索引。

在回调函数中，定义一个空数组 `dataValues` ，用于存储每个多边形的顶点坐标。在 `y.forEach`的循环中根据角度 `angle` 和最大值 `cfg.maxValue` ，计算多边形顶点的 x 坐标 `xCoord` 和 y 坐标 `yCoord` 。

然后使用 `g.append("polygon")` 方法创建一个多边形元素，并绑定数据、样式、鼠标事件。

```js
        .attr("alt", function (j) { return Math.max(j.value, 0) })
        .attr("cx", function (j, i) {
            return cfg.w / 2 * (1 - (Math.max(j.value, 0) / cfg.maxValue) * cfg.factor * Math.sin(i * cfg.radians / total));
        })
        .attr("cy", function (j, i) {
            return cfg.h / 2 * (1 - (Math.max(j.value, 0) / cfg.maxValue) * cfg.factor * Math.cos(i * cfg.radians / total));
        })
        .attr("data-id", function (j) { return j.axis })
        .style("fill", cfg.color(series))
        .style("fill-opacity", .9)
        .on('mouseover', function () {
            g.selectAll("polygon")
                .transition(200)
                .style("fill-opacity", 0.1);
            d3.select(this)
                .transition(200)
                .style("fill-opacity", 0.7);
        })
        .on('mouseout', function () {
            g.selectAll("polygon")
                .transition(200)
                .style("fill-opacity", cfg.opacityArea);
        })
        .append("title")
        .text(function (j) { return Math.max(j.value, 0) });

    series++;
});
```
核心在于根据之前的 series 来知道是第几个围合的多边形，我们可以用于区分样式、事件。

根据之前的 y，我们可以得到每个节点的坐标，绘制 circle 即可。所以，这就是为什么在 `d.forEach` 循环中同时处理多边形和节点。

## 总结

最后，我们总结下实战中要注意的点：

- **D3 动态饼图**：我们从生成环形、标注、引线，我们发现一旦弧线做动画，所有元素都需要过渡，都需要通过 `attrTween` 方法来监听 d、transform、 text-anchor、points。通过 `interpolate` 插值函数和 t 插值来计算出当前的 d、transform、 text-anchor、points。

- **D3 雷达图**：实现雷达图的难点在于每个层级的点的计算，点我们可以通过三角函数得到坐标，另外我们需要乘以强度因子 levelFactor 来做适配。同时由于节点的坐标集合和多边形的坐标集合是一样的，我们可以放在一个循环中执行。
