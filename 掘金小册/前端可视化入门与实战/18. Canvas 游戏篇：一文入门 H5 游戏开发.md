在 Canvas 游戏开发中，**`精灵图`（Sprite Sheet）是一种包含多个小图像帧的图像文件**。

这些小图像帧通常代表了游戏中的不同动作或状态，比如角色的行走、跳跃、攻击等。通过将所有的小图像帧合并到一个精灵图中，可以减少图像加载和渲染的开销，并且方便控制和管理精灵的动画。开发者可以通过指定精灵图中的特定帧来实现动画效果，并在 Canvas 上绘制出相应的精灵。

下面我们就一起来绘制一个动态的精灵图动画：

![image.png](https://p1-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/e7b29a6d1c4c4af498c0ebbf2fec0081~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=650&h=170&s=61546&e=png&b=202020)

绘制核心原理在于 `ctx.drawImage` 控制裁剪第几帧，一旦超过图片宽度就会从 0 帧开始渲染来实现循环播放。通过控制不同精灵图的静态资源路径，我们能模拟不同的动作。为了控制播放速率，我们可以通过一个最大帧率的变量，该变量控制多少渲染帧率移动裁剪一帧。

> [OpenGameArt.org](https://opengameart.org/) 是一个开源游戏艺术资源网站。我们可以获取免费的游戏素材，包括图像、音频、模型等。



## 精灵图绘制器

我们先实现精灵图的绘制器：

```js
 <script>
     class Sprite {
            constructor(imagePath, frameWidth, frameHeight, frameRate) {
                // 创建Canvas元素
                this.canvas = document.createElement('canvas');
                this.ctx = this.canvas.getContext('2d');
                document.body.appendChild(this.canvas);

                // 加载精灵图
                this.spriteImage = new Image();
                this.spriteImage.src = imagePath;
                this.frameWidth = frameWidth;
                this.frameHeight = frameHeight;
                this.frameRate = frameRate;

                this.frameX = 0;
                this.frameY = 0;
                this.currentFrame = 0;

                // 在精灵图加载完成后开始游戏循环
                this.spriteImage.onload = () => {
                    this.canvas.width = this.frameWidth;
                    this.canvas.height = this.frameHeight;
                    this.gameLoop();
                };
            }

            // 绘制精灵图的函数
            drawSprite() {
                this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
                this.ctx.drawImage(
                    this.spriteImage,
                    this.frameX * this.frameWidth,
                    this.frameY * this.frameHeight,
                    this.frameWidth,
                    this.frameHeight,
                    0,
                    0,
                    this.frameWidth,
                    this.frameHeight
                );
            }

            // 更新帧的函数
            updateFrame() {
                this.currentFrame++;
                if (this.currentFrame >= this.frameRate) {
                    this.currentFrame = 0;
                    this.frameX++;
                    if (this.frameX >= Math.floor(this.spriteImage.width / this.frameWidth)) {
                        this.frameX = 0;
                        this.frameY++;
                        if (this.frameY >= this.spriteImage.height / this.frameHeight) {
                            this.frameY = 0;
                        }
                    }
                }
            }

            // 游戏循环，每帧更新和绘制精灵图
            gameLoop() {
                this.updateFrame();
                this.drawSprite();
                requestAnimationFrame(() => this.gameLoop());
            }
        }

        // 创建Sprite实例并传入相关参数
        const sprite = new Sprite('./p1_spritesheet.png', 72.57, 99, 10);
    };
</script>
```
这里需要注意的是：frameY 用来控制 Y 轴的精灵图裁剪。比如，当我们遇到下面这张图的时候，就可以执行 Y 轴的渲染。

## 精灵图组合渲染

实际我们在开发中，不会将动画绘制逻辑跟精灵创建逻辑耦合在一起，所以我们要**抽离一个动画渲染器**，在这个渲染器中我们会维护一个队列，用于组合渲染注册的不同精灵图。

接下来我们来简单实现一个动画渲染器：

```js
class Canvas {
        constructor() {
            this.create()
        }

        create() {
            this.canvas = document.createElement('canvas');
            this.ctx = this.canvas.getContext('2d');
            document.body.appendChild(this.canvas);
            this.canvas.width = 300
            this.canvas.height = 300
        }
    }

class AnimationPlayer extends Canvas {
    imgRenderNum = 0
    renderId = 0
    spriteList = []
    curSprite = null
    frameX = 0
    frameY = 0
    currentFrame = 0
    status = 'play'

    constructor(spriteList) {
        super()
        this.spriteList = spriteList
        this.spriteNum = spriteList.length
        spriteList.forEach(item => {
            item.spriteImage.onload = () => {
                this.imgRenderNum++
                if (this.imgRenderNum === this.spriteNum) {
                    // 此时图片加载完毕
                    this.renderSprite()
                }
            }
        })
    }
    renderSprite() {
        this.curSprite = this.spriteList[this.renderId]
        this.gameLoop()
    }

    // 渲染下一个精灵
    forward_render() {
        this.renderId++
        this.renderSprite()
    }
    // 返回上一个精灵渲染
    back_render() {
        this.renderId = 0
        this.renderSprite()
    }

    // 清理渲染
    clear_render() {
        cancelAnimationFrame(this.animationId)
    }

    pause() {
        this.status === 'pause'
    }

    play() {
        this.status === 'play'
    }

    drawSprite() {
        const { spriteImage, frameWidth, frameHeight } = this.curSprite
        this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        this.ctx.drawImage(
            spriteImage,
            this.frameX * frameWidth,
            this.frameY * frameHeight,
            frameWidth,
            frameHeight,
            0,
            0,
            frameWidth,
            frameHeight
        );
    }

    updateFrame() {
        if (this.status === 'pause') return

        const { spriteImage, frameWidth, frameHeight, frameRate } = this.curSprite
        this.currentFrame++;
        if (this.currentFrame >= frameRate) {
            this.currentFrame = 0;
            this.frameX++;
            if (this.frameX >= Math.floor(spriteImage.width / frameWidth)) {
                this.frameX = 0;
                this.frameY++;

                if (this.frameY >= spriteImage.height / frameHeight) {
                    this.frameY = 0;

                    this.clear_render()
                    if (this.spriteList[this.renderId + 1]) {
                        this.forward_render()
                    } else {
                        this.back_render()
                    }
                }
            }
        }
    }
    gameLoop() {
        this.updateFrame();
        this.drawSprite();
        this.animationId = requestAnimationFrame(() => {
            this.gameLoop()
        });
    }
}

class Sprite {
    constructor(imagePath, frameWidth, frameHeight, frameRate) {
        this.spriteImage = new Image();
        this.spriteImage.src = imagePath;
        this.frameWidth = frameWidth;
        this.frameHeight = frameHeight;
        this.frameRate = frameRate;
        this.currentFrame = 0;
    }
}

const spriteList = [new Sprite('./img/p1_spritesheet.png', 72.57, 99, 16), new Sprite('./img/NewerCampFire_2.png', 64, 64, 16)]

new AnimationPlayer(spriteList)
```


我们专门抽离了画布类、精灵类、动画类方便后期维护，在动画类中我们可以通过 `imgRenderNum` 记录图片的加载状态，加载完成后执行渲染逻辑，通过 `renderId` 记录渲染到第几个精灵图。一旦`spriteList`渲染完后，我们需要清除动画定时器并将 `renderId` 重置，然后循环渲染。

此外，我们还可以拓展出暂停、播放、是否循环等功能。





## 模拟抛物线

在球体弹跳、射击的游戏场景中，我们往往需要模拟物体的抛物线运动，首先我们需要知道抛物线的公式：

```js
x = startX + velocity * t 
y = startY + 0.5 * g * t^2
```

其中，x 和 y 分别表示抛物线上某一点的坐标，startX 和 startY 表示抛物线的起始点的坐标，velocity 表示抛物线的初始速度，g 表示重力加速度，t 表示时间。 

通过这两个公式，可以根据时间 t 的变化计算出抛物线上每个点的 x 和 y 坐标。x 坐标随时间线性增加，y 坐标随时间的平方增加，同时受到重力加速度的影响。

下面是简单的抛物线动画：

```js
const canvas = document.createElement('canvas');
const context = canvas.getContext('2d');
document.body.appendChild(canvas);

canvas.width = window.innerWidth;
canvas.height = window.innerHeight;

const startX = 100; // 起始点 x 坐标
const startY = canvas.height / 2; // 起始点 y 坐标
const g = 0.1; // 重力加速度
const velocity = 5; // 初始速度
let x = startX;
let y = startY;
let t = 0;

function drawBall() {
    context.clearRect(0, 0, canvas.width, canvas.height);
    context.beginPath();
    context.arc(x, y, 10, 0, Math.PI * 2);
    context.fillStyle = 'red';
    context.fill();
    context.closePath();
}

function updatePosition() {
    x = startX + velocity * t;
    y = startY + 0.5 * g * t * t;
    t += 1;

    if (y >= canvas.height) {
        t = 0;
        y = startY;
    }
}

function animate() {
    drawBall();
    updatePosition();
    requestAnimationFrame(animate);
}

animate();
```

## 模拟物体弹跳


```html
<canvas id="canvas" width="600" height="600"></canvas>
<script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    const gravity = 0.5; // 重力加速度
    const friction = 0.8; // 摩擦系数

    class Ball {
        constructor(x, y, radius, color) {
            this.x = x;
            this.y = y;
            this.radius = radius;
            this.color = color;
            this.dy = 0; // 垂直速度
        }

        draw() {
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
            ctx.fillStyle = this.color;
            ctx.fill();
            ctx.closePath();
        }

        update() {
            if (this.y + this.radius + this.dy > canvas.height) {
                this.dy = -this.dy * friction; // 弹跳后的速度反向，并考虑摩擦
            } else {
                this.dy += gravity; // 应用重力加速度
            }

            this.y += this.dy; // 更新垂直位置

            this.draw(); // 绘制球体
        }
    }

    const ball = new Ball(canvas.width / 2, canvas.height / 2, 50, 'red');

    function animate() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ball.update();
        requestAnimationFrame(animate);
    }
    animate();
</script>
```

我们定义了两个常量 `gravity` 和 `friction`，分别表示重力加速度和摩擦系数。

在 `update` 方法中，首先判断球体是否触及画布底部，如果是，则将垂直速度反向并乘以摩擦系数，模拟弹跳效果；否则，应用重力加速度，使球体下落。然后，更新球体的垂直位置。最后，调用 `draw` 方法绘制球体。我们记住下面的公式即可：

```js
// 将速度反向并乘以摩擦系数，模拟反弹效果
this.dy = -this.dy * friction
// 将速度增加重力加速度模拟物体自由落体运动
this.dy += gravity
```

## 碰撞检测

在游戏场景中，我们经常需要判定是否触碰到物体来实现消灭物体、吃到奖励、得分等操作。接下来我们一起看下常用的碰撞检测的方法。



### 外接矩形判别法

**对于两个矩形的碰撞检测**，只要计算坐标是否在另一个矩形的内部，我们这里做个简单的限制就可以：

```js
// 碰撞检测函数
function isColliding(rect1, rect2) {
    if (
        rect1.x < rect2.x + rect2.width &&
        rect1.x + rect1.width > rect2.x &&
        rect1.y < rect2.y + rect2.height &&
        rect1.y + rect1.height > rect2.y
    ) {
        return true;
    }
    return false;
}
```



### 外接圆判别法

**对于两个球体的碰撞检测**，可以通过它们的中心点之间的距离来判断它们是否相交。如果两个球体中心点之间的距离小于它们的半径之和，那么它们发生了碰撞。所以，我们就需要用到勾股定理来计算斜边的长度。

```js
 // 碰撞检测函数
function isColliding(ball1, ball2) {
    const distance = Math.sqrt((ball2.x - ball1.x) ** 2 + (ball2.y - ball1.y) ** 2);
    if (distance < ball1.radius + ball2.radius) {
        return true;
    }
    return false;
}
```




### 光线投射法

简单理解就是计算两条`射线`的交点位置来判断是否碰撞。

**射线是由一个起点和一个方向确定的直线段**。它可以看作是一个无限延伸的直线，但在计算和几何中，通常我们只关注它从起点开始的有限部分。

射线由两个主要属性定义：**起点和方向**。

射线是由某个坐标点有方向性地发射出来，比如人在抛出篮球的时候，球心发出的射线和篮球框发出的射线交点一直在不断接近球筐，这个交点我们可以限制在篮筐以下来保证是否投篮成功。

射线法让我们忽略了物体的形状，根据运动方向考虑物体射线和目标物的射线的相交关系来判断是否碰撞，但是计算大量的方向性的判断就不合适了，**光线投射法更适合物体运动的轨迹行为的一些判定、光照的计算**。

接下来我们通过一个简单的案例来加深理解：

```js
<canvas id="canvas" width="800" height="400"></canvas>

<script>
    // 获取Canvas元素
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    // 定义射线的起点和方向
    const ray = {
        x: 100, // 射线起点的x坐标
        y: 200, // 射线起点的y坐标
        angle: 45, // 射线的角度（以度为单位）
    };
    // 定义墙的位置和尺寸
    const wall = {
        x1: 600, // 墙的起点x坐标
        y1: 100, // 墙的起点y坐标
        x2: 600, // 墙的终点x坐标
        y2: 300, // 墙的终点y坐标
    };

    // 绘制墙
    function drawWall() {
        ctx.strokeStyle = 'gray';
        ctx.beginPath();
        ctx.moveTo(wall.x1, wall.y1);
        ctx.lineTo(wall.x2, wall.y2);
        ctx.stroke();
    }

    // 绘制射线
    function drawRay() {
        ctx.beginPath();
        ctx.moveTo(ray.x, ray.y);
        const angleInRadians = ray.angle * Math.PI / 180;
        // 射线的长度设置为1000个像素
        const endX = ray.x + Math.cos(angleInRadians) * 1000;
        const endY = ray.y + Math.sin(angleInRadians) * 1000;
        // 检测碰撞
        const collision = checkCollision();
        // 根据碰撞结果更新射线的颜色
        if (collision) {
            ctx.strokeStyle = 'red';
        } else {
            ctx.strokeStyle = 'yellow';
        }
        ctx.lineTo(endX, endY);
        ctx.stroke();
    }

    // 渲染函数
    function render() {
        // 清空画布
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // 绘制墙和射线
        drawWall();
        drawRay();

        // 更新射线的角度（可以根据交互或动画来更新）
        ray.angle += 1;

        requestAnimationFrame(render);
    }
    render();
</script>
```

接下来我们看下核心的碰撞检测是如何实现的：

```js
 // 检测射线和墙的碰撞
function checkCollision() {
    const angleInRadians = ray.angle * Math.PI / 180;
    const endX = ray.x + Math.cos(angleInRadians) * 1000; // 射线的长度设置为1000个像素
    const endY = ray.y + Math.sin(angleInRadians) * 1000;

    // 计算墙的射线向量
    const wallVector = {
        x: wall.x2 - wall.x1,
        y: wall.y2 - wall.y1,
    };

    // 计算光的射线向量
    const rayVector = {
        x: endX - ray.x,
        y: endY - ray.y,
    };

    // 判断射线和墙是否相交
    const t1 = (wallVector.x * (ray.y - wall.y1) + wallVector.y * (wall.x1 - ray.x)) / (rayVector.x * wallVector.y - rayVector.y * wallVector.x);
    const t2 = (rayVector.x * (wall.y1 - ray.y) + rayVector.y * (ray.x - wall.x1)) / (wallVector.x * rayVector.y - wallVector.y * rayVector.x);

    if (t1 >= 0 && t1 <= 1 && t2 >= 0 && t2 <= 1) {
        console.log("碰撞发生");
        return true;
    }
    console.log("没有碰撞");

    return false;
}
```

接下来推导射线相交的原理：我们可以将墙的起点坐标表示为 `(wall.x1, wall.y1)` ，射线的起点坐标表示为 `(ray.x, ray.y)` ，交点的坐标表示为 `(ray.x + t1 * rayVector.x, ray.y + t1 * rayVector.y)` ，这里 rayVector 的 x 和 y 可以简单理解为首尾线段坐标的间距，t1 和 t2 代表线段的倍数。

![动画.apng](https://p6-juejin.byteimg.com/tos-cn-i-k3u1fbpfcp/9ec51a8cebfe4047aa2d90e4d75029eb~tplv-k3u1fbpfcp-jj-mark:0:0:0:0:q75.image#?w=542&h=358&s=316564&e=apng&f=150&b=ffffff)

一旦线性相交会满足这个条件：

```js
(ray.x + t1 * rayVector.x, ray.y + t1 * rayVector.y) =
(wall.x1 + t2 * wallVector.x, wall.y1 + t2 * wallVector.y)
```

然后我们可以得到两个等式：

```js
ray.x + t1 * rayVector.x = wall.x1 + t2 * wallVector.x 
ray.y + t1 * rayVector.y = wall.y1 + t2 * wallVector.y
```

接下来，我们使用克拉默法则解这个线性方程组，计算 `t1` 和 `t2` 的值。

```js
t1 = (wallVector.x * (ray.y - wall.y1) + wallVector.y * (wall.x1 - ray.x)) / (rayVector.x * wallVector.y - rayVector.y * wallVector.x)
t2 = (rayVector.x * (wall.y1 - ray.y) + rayVector.y * (ray.x - wall.x1)) / (wallVector.x * rayVector.y - wallVector.y * rayVector.x)
```


通过计算 `t1` 和 `t2` 的值，我们可以判断射线和墙是否相交。如果 `t1` 和 `t2` 的值都在 0 到 1 之间，表示相交点在射线和墙的线段上，即射线和墙相交。

射线可以方便地检测任意多边形是否相交，并更适合快速移动不断计算的物体，而边界判定的方法是不适合任意多边形的。

接下来我们将介绍 `SAT` 和 `MTV` 两种基于分割轴算法来实现`更精确的碰撞检测`。




### 分离轴定理（SAT）

分离轴定理是一种用于检测凸多边形或凸多边形与其他凸多边形或凸包之间是否相交的方法。

它基于一个简单的原理：如果两个凸多边形不相交，那么一定存在一条直线（即**分离轴**），可以将它们完全分开。换句话说，两个凸多边形之间的分离轴是不存在重叠的。通过检查两个凸多边形是否存在重叠的分离轴，我们可以判断它们是否相交。

概念可能有点难理解，接下来我演示几张图你就明白了：

首先多边形外围的所有线段都为投影轴，轴上的为多边形投射过来的投影。投影轴的数量根据受测多边形的边长决定。一旦存在投影重叠，那么意味着两个多边形是碰撞了。就像右图的投影轴 2 上存在了重叠。

当然，如果多边形的边长很多，计算相比投射更消耗性能，因为投影轴更多。如果我们一旦检测到重叠时，后续的计算就没有必要做了。

由于 Web 端是没有原生封装 Vector 向量类，接下来我们要手写实现：

```js
const Vector = function(x, y) {
  this.x = x;
  this.y = y;
};

Vector.prototype = {
  // 计算向量的长度（模）
  getMagnitude: function() {
    return Math.sqrt(this.x * this.x + this.y * this.y);
  },
  // 向量相加
  add: function(vector) {
    return new Vector(this.x + vector.x, this.y + vector.y);
  },
  // 向量相减
  subtract: function(vector) {
    return new Vector(this.x - vector.x, this.y - vector.y);
  },
  // 向量的点积
  dotProduct: function(vector) {
    return this.x * vector.x + this.y * vector.y;
  },
  // 计算两个向量形成的边向量
  edge: function(vector) {
    return this.subtract(vector);
  },
  // 计算向量的垂直向量
  perpendicular: function() {
    return new Vector(-this.y, this.x);
  },
  // 向量归一化为单位向量
  normalize: function() {
    const magnitude = this.getMagnitude();
    return new Vector(this.x / magnitude, this.y / magnitude);
  },
  // 计算向量的单位法向量
  normal: function() {
    return this.perpendicular().normalize();
  }
};
```

接着我们实现核心的 **`checkCollision`（SAT 算法）方法**：


```js
function checkCollision() {
  // 获取四边形和六边形的边
  const quadEdges = getEdges(quadPoints);
  const hexEdges = getEdges(hexPoints);

  // 获取四边形和六边形的法向量
  const quadNormals = getNormals(quadEdges);
  const hexNormals = getNormals(hexEdges);

  // 检查四边形和六边形的所有边是否分离
  if (isSeparating(quadPoints, hexPoints, quadNormals.concat(hexNormals))) {
    return false; // 分离，没有碰撞
  }
  // 没有找到分离轴，发生碰撞
  return true;
}

// 获取多边形的边
function getEdges(points) {
  const edges = [];
  for (let i = 0; i < points.length; i++) {
    const p1 = points[i];
    const p2 = points[(i + 1) % points.length];
    edges.push({ x: p2.x - p1.x, y: p2.y - p1.y });
  }
  return edges;
}

// 获取边的法向量
function getNormals(edges) {
  const normals = [];
  for (const edge of edges) {
    const normal = { x: -edge.y, y: edge.x };
    normals.push(normalize(normal));
  }
  return normals;
}

// 归一化向量
function normalize(vector) {
  const magnitude = Math.sqrt(vector.x * vector.x + vector.y * vector.y);
  return { x: vector.x / magnitude, y: vector.y / magnitude };
}

// 检查两个多边形是否分离
function isSeparating(points1, points2, axes) {
  for (const axis of axes) {
    const projection1 = project(points1, axis);
    const projection2 = project(points2, axis);
    if (!overlap(projection1, projection2)) {
      return true; // 分离，没有碰撞
    }
  }
  return false; // 没有找到分离轴，有碰撞
}

// 在给定轴上投影多边形
function project(points, axis) {
  let min = Infinity;
  let max = -Infinity;
  for (const point of points) {
    const projection = point.x * axis.x + point.y * axis.y;
    min = Math.min(min, projection);
    max = Math.max(max, projection);
  }
  return { min, max };
}

// 检查两个投影是否重叠
function overlap(projection1, projection2) {
  return (
    (projection1.min <= projection2.max && projection1.max >= projection2.min) ||
    (projection2.min <= projection1.max && projection2.max >= projection1.min)
  );
}
```

这样我们就实现了 Canvas 的碰撞检测，**在 H5 的游戏开发中对于复杂多边形的检测，可以直接套用我们这里封装的 SAT 算法**。


### 最小平移向量（MTV）

最小平移向量是指在碰撞检测中，两个物体发生碰撞时，需要将它们分离的**最小平移距离和方向**。当两个物体发生碰撞时，我们可以计算出碰撞产生的**重叠区域**，并通过计算最小平移向量来移动其中一个物体，使它们不再重叠。最小平移向量通常用于解决`碰撞响应`和`碰撞恢复`的问题。

实现原理也很简单，跟之前 SAT 算法一样得到投影轴，然后计算两个受测多边形的投影长度，得到重叠区域，再将所有投影轴的重叠区域做比较取最小的平移向量即可。


下面演示具体的案例代码：


```js
<!DOCTYPE html>
<html>

<head>
    <style>
        canvas {
            border: 1px solid black;
        }
    </style>
</head>

<body>
    <canvas id="canvas" width="400" height="400"></canvas>

    <script>
        // 获取 Canvas 元素和上下文
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');

        // 定义多边形的顶点坐标
        const polygon1 = [
            { x: 100, y: 100 },
            { x: 200, y: 100 },
            { x: 200, y: 200 },
            { x: 100, y: 200 }
        ];

        const polygon2 = [
            { x: 150, y: 150 },
            { x: 100, y: 100 },
            { x: 350, y: 150 },
            { x: 350, y: 250 },
            { x: 300, y: 300 },
            { x: 250, y: 250 }
        ];

        // 绘制多边形
        function drawPolygon(points, color) {
            ctx.beginPath();
            ctx.moveTo(points[0].x, points[0].y);
            for (let i = 1; i < points.length; i++) {
                ctx.lineTo(points[i].x, points[i].y);
            }
            ctx.closePath();
            ctx.fillStyle = color;
            ctx.fill();
        }

        // 计算两个多边形之间的最小平移向量
        function calculateMTV(polygon1, polygon2) {
            let minOverlap = Infinity;
            let smallestAxis = null;

            // 获取多边形的边和法向量
            const axes = getAxes(polygon1.concat(polygon2));

            for (const axis of axes) {
                // 投影1
                const projection1 = project(polygon1, axis);
                // 投影2
                const projection2 = project(polygon2, axis);

                // 计算投影的重叠量
                const overlap = getOverlap(projection1, projection2);

                if (overlap === 0) {
                    // 发生分离，没有碰撞
                    return null;
                } else {
                    // 更新最小重叠量和对应的法向量
                    if (overlap < minOverlap) {
                        minOverlap = overlap;
                        smallestAxis = axis;
                    }
                }
            }

            // 返回最小平移向量
            return { x: smallestAxis.x * minOverlap, y: smallestAxis.y * minOverlap };
        }

        // 获取多边形的边和法向量
        function getAxes(points) {
            const axes = [];
            for (let i = 0; i < points.length; i++) {
                const p1 = points[i];
                const p2 = points[(i + 1) % points.length];
                const edge = { x: p2.x - p1.x, y: p2.y - p1.y };
                const axis = { x: -edge.y, y: edge.x };
                const normalizedAxis = normalize(axis);
                axes.push(normalizedAxis);
            }
            return axes;
        }

        // 归一化向量
        function normalize(vector) {
            const magnitude = Math.sqrt(vector.x * vector.x + vector.y * vector.y);
            return { x: vector.x / magnitude, y: vector.y / magnitude };
        }

        // 在给定轴上投影多边形
        function project(points, axis) {
            let min = Infinity;
            let max = -Infinity;
            for (const point of points) {
                const projection = point.x * axis.x + point.y * axis.y;
                min = Math.min(min, projection);
                max = Math.max(max, projection);
            }
            return { min, max };
        }

        // 计算投影的重叠量
        function getOverlap(projection1, projection2) {
            return Math.min(projection1.max, projection2.max) - Math.max(projection1.min, projection2.min);
        }

        // 绘制场景
        function drawScene() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // 绘制多边形1
            drawPolygon(polygon1, 'blue');

            // 绘制多边形2
            drawPolygon(polygon2, 'green');

            // 计算最小平移向量
            const mtv = calculateMTV(polygon1, polygon2);

            // 可视化绘制演示
            if (mtv) {
                // 绘制最小平移向量
                ctx.beginPath();
                ctx.moveTo(polygon2[0].x, polygon2[0].y);
                ctx.lineTo(polygon2[0].x + mtv.x, polygon2[0].y + mtv.y);
                ctx.strokeStyle = 'red';
                ctx.lineWidth = 2;
                ctx.stroke();
            }
        }

        // 更新画面
        function update() {
            drawScene();
            requestAnimationFrame(update);
        }

        // 启动动画循环
        update();
    </script>
</body>

</html>
```

## 点选

在 SVG 中我们可以直接绑定事件，但是在 Canvas 中我们是无法绑定事件的，所以我们需要用其他方法来实现点击选中的交互。最简单的莫过于判断鼠标是否在矩形判定框内，但是大部分人不知道的是我们还可以用`颜色检测方法`实现。

颜色检测方法，我们会单独创建一个canvas上下文，给每个物体设置唯一的颜色标识，就是为每个物体分配一个唯一的 RGB 颜色值的矩形框。监听鼠标点击事件或其他交互事件，获取点击位置的像素颜色。根据点击位置的像素颜色，匹配对应的物体标识颜色，从而确定点击的是哪个物体。

```js
// 假设有两个物体，分别用红色和绿色表示
var object1Color = [255, 0, 0]; // 红色
var object2Color = [0, 255, 0]; // 绿色

// 绘制场景
function drawScene() {
  var canvas = document.getElementById('canvas');
  var ctx = canvas.getContext('2d');

  // 绘制物体1
  ctx.fillStyle = 'rgb(' + object1Color.join(',') + ')';
  ctx.fillRect(50, 50, 100, 100);

  // 绘制物体2
  ctx.fillStyle = 'rgb(' + object2Color.join(',') + ')';
  ctx.fillRect(200, 50, 100, 100);
}

// 监听鼠标点击事件
canvas.addEventListener('click', function(event) {
  var rect = canvas.getBoundingClientRect();
  var x = event.clientX - rect.left;
  var y = event.clientY - rect.top;

  // 获取点击位置的像素颜色
  var ctx = canvas.getContext('2d');
  var pixelData = ctx.getImageData(x, y, 1, 1).data;
  
  // 根据像素颜色匹配物体
  if (compareColors(pixelData, object1Color)) {
    // 点击了物体1
    console.log('点击了物体1');
  } else if (compareColors(pixelData, object2Color)) {
    // 点击了物体2
    console.log('点击了物体2');
  }
});

// 比较两个颜色是否相等
function compareColors(color1, color2) {
  return color1[0] === color2[0] &&
         color1[1] === color2[1] &&
         color1[2] === color2[2];
}

// 渲染场景
drawScene();
```
